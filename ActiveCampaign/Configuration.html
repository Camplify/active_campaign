def initialize
  self.adapter         = :net_http
  self.api_url         = API_URL
  self.api_timeout     = 5
  self.api_token       = API_TOKEN
  self.debug           = false
  self.logger          = Logger.new(STDOUT)
  @request_middleware  = {}
  @response_middleware = {}
end
